function googleDocCallback(data) { if (!data || !data.table) return; const rows = data.table.rows; const grid = document.getElementById('main-grid'); grid.innerHTML = ''; const now = new Date(); const currentMinutes = now.getHours() * 60 + now.getMinutes(); // זמן נוכחי בדקות let targetSessionRows = []; let currentScanningSession = ""; // שלב 1: מציאת האימון הרלוונטי לפי השעה for (let i = 0; i < rows.length; i++) { const firstCell = rows[i].c[0] ? rows[i].c[0].v.toString().trim() : ""; if (firstCell.includes("אימון")) { const timeMatch = firstCell.match(/(\d{2}):(\d{2})/); if (timeMatch) { const sessionHour = parseInt(timeMatch[1]); const sessionMin = parseInt(timeMatch[2]); const sessionMinutes = sessionHour * 60 + sessionMin; // בדיקה אם זה האימון האחרון בגיליון let isLast = true; for (let k = i + 1; k < rows.length; k++) { if (rows[k].c[0] && rows[k].c[0].v.toString().includes("אימון")) { isLast = false; break; } } // תנאי תצוגה: 10 דקות לפני, ועד שהבא מתחיל (או 70 דק' אם זה האחרון) const startTime = sessionMinutes - 10; const endTime = isLast ? sessionMinutes + 70 : sessionMinutes + 60; // הנחה שאימון רגיל מתחלף אחרי שעה if (currentMinutes >= startTime && currentMinutes < endTime) { currentScanningSession = firstCell; targetSessionRows = []; } else { currentScanningSession = ""; } } continue; } // שלב 2: איסוף המתאמנות של האימון שנמצא if (currentScanningSession !== "" && firstCell !== "" && isNaN(firstCell)) { targetSessionRows.push(rows[i]); } } // שלב 3: הצגה על המסך if (targetSessionRows.length > 0) { targetSessionRows.slice(0, 6).forEach(row => { const name = row.c[0].v; let html = `
${name}
`; for (let j = 1; j < row.c.length; j += 3) { const ex = row.c[j] ? row.c[j].v : ''; const weight = (row.c[j+1] && row.c[j+1].v) ? row.c[j+1].v : ''; const reps = (row.c[j+2] && row.c[j+2].v) ? row.c[j+2].v : '-'; if (ex && ex !== "" && ex !== "...") { html += `
${ex} ${weight ? `${weight}` : ''}
${reps}
`; } } html += `
`; grid.innerHTML += html; }); } }
