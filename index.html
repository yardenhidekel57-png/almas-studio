function renderWorkoutCards(rows) { const grid = document.getElementById('main-grid'); let targetRows = []; let activeTitle = "אימון פעיל"; // חיפוש האימון האחרון שמופיע בגיליון (בלי קשר לשעה) for (let i = 0; i < rows.length; i++) { const firstCell = rows[i].c[0] ? rows[i].c[0].v.toString().trim() : ""; if (firstCell.includes("אימון")) { activeTitle = firstCell; targetRows = []; // מאפס כדי לקחת את האימון הכי עדכני שמצאנו let j = i + 1; while (j < rows.length && rows[j].c[0] && !rows[j].c[0].v.includes("אימון")) { targetRows.push(rows[j]); j++; } } } if (targetRows.length > 0) { document.getElementById('session-info').innerText = activeTitle; grid.innerHTML = targetRows.map(row => { const name = row.c[0].v; let exerciseHtml = ''; for (let k = 1; k < row.c.length; k += 3) { const ex = row.c[k] ? row.c[k].v : ''; const weight = (row.c[k+1] && row.c[k+1].v) ? row.c[k+1].v : ''; const reps = (row.c[k+2] && row.c[k+2].v) ? row.c[k+2].v : '-'; if (ex && ex !== "...") { exerciseHtml += `
${ex} ${weight}
${reps}
`; } } return `
${name}
${exerciseHtml}
`; }).join(''); } else { document.getElementById('session-info').innerText = "מוכן לאימון"; grid.innerHTML = '
ALMA STUDIO
'; } }
