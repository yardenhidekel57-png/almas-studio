const SHEET_ID = '18DFppHukNFORCDMr4CAyZ-LlEOgBKVDbliLN4-vpFCk'; function parseSheetTime(str) { const match = str.match(/(\d{1,2}):(\d{2})/); if (!match) return null; const d = new Date(); d.setHours(parseInt(match[1]), parseInt(match[2]), 0, 0); return d; } function handleData(data) { if (!data || !data.table) return; const rows = data.table.rows; const now = new Date(); let sessions = []; let currentSession = null; // חילוץ האימונים מהגיליון for (let i = 0; i < rows.length; i++) { const cellVal = (rows[i].c && rows[i].c[0]) ? rows[i].c[0].v : ""; if (typeof cellVal === 'string' && cellVal.includes("אימון")) { if (currentSession) sessions.push(currentSession); currentSession = { time: parseSheetTime(cellVal), title: cellVal, trainees: [] }; } else if (currentSession && cellVal && cellVal !== "מתאמנת") { currentSession.trainees.push(rows[i].c); } } if (currentSession) sessions.push(currentSession); // לוגיקת הצגה: בדיוק שעה מתחילת האימון let activeSession = null; for (let i = 0; i < sessions.length; i++) { const s = sessions[i]; const nextS = sessions[i+1]; if (!s.time) continue; const startTime = s.time; // זמן סיום: או האימון הבא, או בדיוק 60 דקות אחרי תחילת הנוכחי const endTime = nextS ? nextS.time : new Date(s.time.getTime() + 60 * 60000); if (now >= startTime && now < endTime) { activeSession = s; break; } } renderView(activeSession); } function renderView(session) { const container = document.getElementById('container'); const info = document.getElementById('training-info'); if (!session || session.trainees.length === 0) { container.innerHTML = '
אין אימון פעיל כרגע
'; info.innerText = 'הפסקה'; return; } info.innerText = session.title; const trainees = session.trainees; container.innerHTML = ''; const numTrainees = trainees.length; const maxPerRow = numTrainees <= 3 ? numTrainees : Math.ceil(numTrainees / 2); let rowDiv = document.createElement('div'); rowDiv.className = 'row'; trainees.forEach((trainee, index) => { if (index > 0 && index % maxPerRow === 0) { container.appendChild(rowDiv); rowDiv = document.createElement('div'); rowDiv.className = 'row'; } const name = trainee[0] ? trainee[0].v : "מתאמנת"; let exercisesHtml = ""; for (let j = 1; j < trainee.length; j += 3) { const ex = trainee[j] ? trainee[j].v : ''; if (ex && ex !== "" && ex !== "...") { const w = (trainee[j+1] && trainee[j+1].v) ? trainee[j+1].v : ''; const r = (trainee[j+2] && trainee[j+2].v) ? trainee[j+2].v : ''; exercisesHtml += `
${ex}${w}
${r}
`; } } rowDiv.innerHTML += `
${name}
${exercisesHtml}
`; }); container.appendChild(rowDiv); } function refresh() { const s = document.createElement('script'); s.src = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=responseHandler:handleData&gid=0&tcb=${Date.now()}`; document.head.appendChild(s); s.onload = () => s.remove(); } setInterval(refresh, 10000); setInterval(() => { document.getElementById('time').innerText = new Date().toLocaleTimeString('he-IL', {hour12:false}); }, 1000); refresh();
